		if (key > node->kv.first) return search(node->right, std::move(key));

	pointer maximum(pointer x) {
		if (x->right == nil) return x;
		return maximum(x->right);
	}

	pointer minimum(pointer x) {
		if (x->left == nil) return x;
		return minimum(x->left);

	void transplant(pointer u, pointer v) {

		// æŒ‡é’ˆæ”¹å˜æŒ‡å‘
		if (u->parent == nil) root = v;
		else if (u == u->parent->left) u->parent->left = v;
		else u->parent->right = v;
		v->parent = u->parent;

	}

	void remove(pointer z) {

		pointer x = nil;
		pointer y = z;
		Color ycolor = y->color;

		// zåªæœ‰å³å­©å­
		if (z->left = nil) {
			x = z->right;
			transplant(z, z->right);
		}
		// zåªæœ‰å·¦å­©å­
		else if (z->right == nil) {
			x = z->left;
			transplant(z, z->left);
		}
		else {
			y = minimum(z->right);
			ycolor = y->color;
			x = y->right;
			if (y->parent == z) x->parent = y;
			else {
				transplant(y, y->right);
				y->right = z->right;
				y->right->parent = y;
			}
			transplant(z, y);

			// æ”¹å˜æŒ‡å‘
			y->left = z->left;
			z->left->parent = y;
			y->color = z->color;
		}

		if (ycolor == Color::BLACK) remove_fixed(x);

	}
